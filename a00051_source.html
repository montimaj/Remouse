3 /**
4  * Class to represent a quaternion.
5  *
6  * <p>
7  * A quaternion <i>H</i> may be identified with R<sup>4</sup>,
8  * a four-dimensional vector space over the real numbers. The
9  * elements of this basis are customarily denoted as <i>1</i>,
10  * <i>i</i>, <i>j</i>, and <i>k</i>. Every element of <i>H</i>
11  * can be uniquely written as a linear combination of these
12  * basis elements, that is, as <i>a1 + bi + cj + dk</i>, where
13  * <i>a</i>, <i>b</i>, <i>c</i>, and <i>d</i> are real
14  * numbers.
15  * </p>
16  * <p>
17  * Quaternions allow for elegant descriptions of 3D rotations,
18  * interpolations as well as extrapolations and compared to
19  * Euler angles, they don't suffer from gimbal lock. I
20  * nterpolations between two Quaternions are called SLERP
21  * (Spherical Linear Interpolation).
22  * </p>
23  * <p>
24  * If <i>a + bi + cj + dk</i> is any quaternion, then <i>a</i>
25  * is called its scalar part and <i>bi + cj + dk</i> is called
26  * its vector part. The scalar part of a quaternion is always
27  * real, and the vector part is always pure imaginary.
28  * </p>
29  * <p>
30  * Using the basis <i>1, i, j, k</i> of H it is possible to
31  * write <i>H</i> as a set of quadruples:
32  * </p>
33  * <p>
34  * &nbsp;&nbsp;&nbsp;&nbsp;
35  * <b><i>H = {(a,b,c,d) | a,b,c,d &isin; R}</i></b>.
36  * </p>
37  * <p>
38  * Then the basis elements are: <br/>
39  * <i>1 = (1,0,0,0)</i>, <br/>
40  * <i>i = (0,1,0,0)</i>, <br/>
41  * <i>j = (0,0,1,0)</i>, <br/>
42  * <i>k = (0,0,0,1)</i>. <br/>
43  * </p>
44  * <p>
45  * <b>Addition of two quaternions : </b><br/>
46  * <i>
47  * (a<sub>1</sub>,b<sub>1</sub>,c<sub>1</sub>,d<sub>1</sub>) +
48  * (a<sub>2</sub>,b<sub>2</sub>,c<sub>2</sub>,d<sub>2</sub>) =
49  * (a<sub>1</sub>+a<sub>2</sub>, b<sub>1</sub>+b<sub>2</sub>,
50  * c<sub>1</sub>+c<sub>2</sub>, d<sub>1</sub>+d<sub>2</sub>)
51  * </i>.
52  * </p>
53  * <p>
54  * <b>Multiplications of two quaternions : </b><br/>
55  * <i>
56  * (a<sub>1</sub>,b<sub>1</sub>,c<sub>1</sub>,d<sub>1</sub>)
57  * (a<sub>2</sub>,b<sub>2</sub>,c<sub>2</sub>,d<sub>2</sub>) =
58  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
59  * (a<sub>1</sub>a<sub>2</sub> - b<sub>1</sub>b<sub>2</sub> -
60  * c<sub>1</sub>c<sub>2</sub> - d<sub>1</sub>d<sub>2</sub>,
61  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
62  * a<sub>1</sub>b<sub>2</sub> + b<sub>1</sub>a<sub>2</sub> +
63  * c<sub>1</sub>d<sub>2</sub> - d<sub>1</sub>c<sub>2</sub>,
64  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
65  * a<sub>1</sub>c<sub>2</sub> - b<sub>1</sub>d<sub>2</sub> +
66  * c<sub>1</sub>a<sub>2</sub> + d<sub>1</sub>b<sub>2</sub>,
67  * <br/> &nbsp;&nbsp;&nbsp;&nbsp;
68  * a<sub>1</sub>d<sub>2</sub> + b<sub>1</sub>c<sub>2</sub> -
69  * c<sub>1</sub>b<sub>2</sub> + d<sub>1</sub>a<sub>2</sub>)
70  * </i>.
71  * </p>
72  * <p>
73  * <b>Normalization of quaternion</b> <i>(a,b,c,d)</i> <br/>
74  * &nbsp;&nbsp;&nbsp;&nbsp;
75  * = <i>(a/mag, b/mag, c/mag, d/mag)</i>, <br/>
76  * where, <br/> &nbsp;&nbsp;&nbsp;&nbsp;
77  * <i>mag =
78  * &radic;(a<sup>2</sup>+b<sup>2</sup>+c<sup>2</sup>+d<sup>2</sup>)</i>.
79  * </p>
80  * <p>
81  * The sensor data for 3-D mouse movement is sent from the
82  * mobile device is represented in the form of a
83  * <code>Quaternion</code>.
84  * </p>
85  *
86  * @see project.android.sensor.representation.Vector4f
87  */

303  private void generateQuaternionFromMatrix() {
304 
305  float qx;
306  float qy;
307  float qz;
308  float qw;
309 
310  float[] mat = mMatrix.getMatrix();
311  int[] indices;
312 
313  if (this.mMatrix.size() == 16) {
314  if (this.mMatrix.isColumnMajor()) {
315  indices = MatrixF4x4.MAT_IND_COL16_3X3;
316  } else {
317  indices = MatrixF4x4.MAT_IND_ROW16_3X3;
318  }
319  } else {
320  if (this.mMatrix.isColumnMajor()) {
321  indices = MatrixF4x4.MAT_IND_COL9_3X3;
322  } else {
323  indices = MatrixF4x4.MAT_IND_ROW9_3X3;
324  }
325  }
326 
327  int m00 = indices[0];
328  int m01 = indices[1];
329  int m02 = indices[2];
330 
331  int m10 = indices[3];
332  int m11 = indices[4];
333  int m12 = indices[5];
334 
335  int m20 = indices[6];
336  int m21 = indices[7];
337  int m22 = indices[8];
338 
339  float tr = mat[m00] + mat[m11] + mat[m22];
340  if (tr > 0) {
341  float s = (float) Math.sqrt(tr + 1.0) * 2;
342  qw = 0.25f * s;
343  qx = (mat[m21] - mat[m12]) / s;
344  qy = (mat[m02] - mat[m20]) / s;
345  qz = (mat[m10] - mat[m01]) / s;
346  } else if ((mat[m00] > mat[m11]) & (mat[m00] > mat[m22])) {
347  float s = (float) Math.sqrt(1.0 + mat[m00] - mat[m11] - mat[m22]) * 2;
348  qw = (mat[m21] - mat[m12]) / s;
349  qx = 0.25f * s;
350  qy = (mat[m01] + mat[m10]) / s;
351  qz = (mat[m02] + mat[m20]) / s;
352  } else if (mat[m11] > mat[m22]) {
353  float s = (float) Math.sqrt(1.0 + mat[m11] - mat[m00] - mat[m22]) * 2;
354  qw = (mat[m02] - mat[m20]) / s;
355  qx = (mat[m01] + mat[m10]) / s;
356  qy = 0.25f * s;
357  qz = (mat[m12] + mat[m21]) / s;
358  } else {
359  float s = (float) Math.sqrt(1.0 + mat[m22] - mat[m00] - mat[m11]) * 2;
360  qw = (mat[m10] - mat[m01]) / s;
361  qx = (mat[m02] + mat[m20]) / s;
362  qy = (mat[m12] + mat[m21]) / s;
363  qz = 0.25f * s;
364  }
365 
366  setX(qx);
367  setY(qy);
368  setZ(qz);
369  setW(qw);
370  }

404  setY((float) (s1 * c2 * c3 + c1 * s2 * s3));
405  setZ((float) (c1 * s2 * c3 - s1 * c2 * s3));

442  /**
443  * Computes a linear interpolation between this <code>Quaternion</code> and the
444  * input <code>Quaternion</code>.
445  *
446  * @param input the <code>Quaternion</code> to be slerped with this.
447  * @param output the <code>Quaternion</code> to store the result.
448  * @param t the ratio between the two quaternions where 0 <= t <= 1.0 . <br/>
449  * <b><i>Note:</i></b> Increase value of <code>t</code> will bring
450  * rotation closer to the input <code>Quaternion</code>.
451  */
452  public void slerp(Quaternion input, Quaternion output, float t) {
453  // Calculate angle between them
454  Quaternion bufferQuat;
455  float cosHalftheta = this.dotProduct(input);
456 
457  if (cosHalftheta < 0) {
458  if(mTmpQuaternion == null) mTmpQuaternion = new Quaternion();
459  bufferQuat = mTmpQuaternion;
460  cosHalftheta = -cosHalftheta;
461  bufferQuat.mPoints[0] = (-input.mPoints[0]);
462  bufferQuat.mPoints[1] = (-input.mPoints[1]);
463  bufferQuat.mPoints[2] = (-input.mPoints[2]);
464  bufferQuat.mPoints[3] = (-input.mPoints[3]);
465  } else {
466  bufferQuat = input;
467  }
468 
469  if (Math.abs(cosHalftheta) >= 1.0) {
470  output.mPoints[0] = (this.mPoints[0]);
471  output.mPoints[1] = (this.mPoints[1]);
472  output.mPoints[2] = (this.mPoints[2]);
473  output.mPoints[3] = (this.mPoints[3]);
474  } else {
475  double sinHalfTheta = Math.sqrt(1.0 - cosHalftheta * cosHalftheta);
476  double halfTheta = Math.acos(cosHalftheta);
477 
478  double ratioA = Math.sin((1 - t) * halfTheta) / sinHalfTheta;
479  double ratioB = Math.sin(t * halfTheta) / sinHalfTheta;
480 
481  //Calculate Quaternion
482  output.mPoints[3] = ((float) (mPoints[3] * ratioA + bufferQuat.mPoints[3] * ratioB));
483  output.mPoints[0] = ((float) (this.mPoints[0] * ratioA + bufferQuat.mPoints[0] * ratioB));
484  output.mPoints[1] = ((float) (this.mPoints[1] * ratioA + bufferQuat.mPoints[1] * ratioB));
485  output.mPoints[2] = ((float) (this.mPoints[2] * ratioA + bufferQuat.mPoints[2] * ratioB));
486  }
487  }

489  /**
490  * Rotates a 3-D vector by this <code>Quaternion</code>.
491  *
492  * @param v the {@link Vector3f} to be rotated.
493  * @return the rotated {@link Vector3f}.
494  */
495  public Vector3f rotateVector(Vector3f v) {
496  float q0 = this.mPoints[3];
497  float q1 = this.mPoints[0];
498  float q2 = this.mPoints[1];
499  float q3 = this.mPoints[2];
500  float v1 = v.getX();
501  float v2 = v.getY();
502  float v3 = v.getZ();
503  float x = (1-2*q2*q2-2*q3*q3)*v1 + 2*(q1*q2+q0*q3)*v2 + 2*(q1*q3-q0*q2)*v3;
504  float y = 2*(q1*q2-q0*q3)*v1 + (1-2*q1*q1-2*q3*q3)*v2 + 2*(q2*q3+q0*q1)*v3;
505  float z = 2*(q1*q3-q0*q2) + (q2*q3-q0*q1)*v2 + (1-2*q1*q1-2*q2*q2)*v3;
506  return new Vector3f(x,y,z);
507  }
