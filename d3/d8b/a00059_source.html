144  /**
145  * Inverts a 4 x 4 matrix.
146  *
147  * @param mInv the array that holds the output inverted matrix.
148  * @param mInvOffset an offset into <code>mInv</code> where the inverted
149  * matrix is stored.
150  * @param m the input array.
151  * @param mOffset an offset into <code>m</code> where the matrix is stored.
152  * @return <code>true</code>, if the matrix could be inverted, <br/>
153  * <code>false</code>, otherwise.
154  */
155  public static boolean invertM(float[] mInv, int mInvOffset, float[] m, int mOffset) {
156  // Invert a 4 x 4 matrix using Cramer's Rule
157 
158  // transpose matrix
159  final float src0 = m[mOffset];
160  final float src4 = m[mOffset + 1];
161  final float src8 = m[mOffset + 2];
162  final float src12 = m[mOffset + 3];
163 
164  final float src1 = m[mOffset + 4];
165  final float src5 = m[mOffset + 5];
166  final float src9 = m[mOffset + 6];
167  final float src13 = m[mOffset + 7];
168 
169  final float src2 = m[mOffset + 8];
170  final float src6 = m[mOffset + 9];
171  final float src10 = m[mOffset + 10];
172  final float src14 = m[mOffset + 11];
173 
174  final float src3 = m[mOffset + 12];
175  final float src7 = m[mOffset + 13];
176  final float src11 = m[mOffset + 14];
177  final float src15 = m[mOffset + 15];
178 
179  // calculate pairs for first 8 elements (cofactors)
180  final float atmp0 = src10 * src15;
181  final float atmp1 = src11 * src14;
182  final float atmp2 = src9 * src15;
183  final float atmp3 = src11 * src13;
184  final float atmp4 = src9 * src14;
185  final float atmp5 = src10 * src13;
186  final float atmp6 = src8 * src15;
187  final float atmp7 = src11 * src12;
188  final float atmp8 = src8 * src14;
189  final float atmp9 = src10 * src12;
190  final float atmp10 = src8 * src13;
191  final float atmp11 = src9 * src12;
192 
193  // calculate first 8 elements (cofactors)
194  final float dst0 = (atmp0 * src5 + atmp3 * src6 + atmp4 * src7) - (atmp1 * src5 + atmp2 * src6 + atmp5 * src7);
195  final float dst1 = (atmp1 * src4 + atmp6 * src6 + atmp9 * src7) - (atmp0 * src4 + atmp7 * src6 + atmp8 * src7);
196  final float dst2 = (atmp2 * src4 + atmp7 * src5 + atmp10 * src7)
197  - (atmp3 * src4 + atmp6 * src5 + atmp11 * src7);
198  final float dst3 = (atmp5 * src4 + atmp8 * src5 + atmp11 * src6)
199  - (atmp4 * src4 + atmp9 * src5 + atmp10 * src6);
200  final float dst4 = (atmp1 * src1 + atmp2 * src2 + atmp5 * src3) - (atmp0 * src1 + atmp3 * src2 + atmp4 * src3);
201  final float dst5 = (atmp0 * src0 + atmp7 * src2 + atmp8 * src3) - (atmp1 * src0 + atmp6 * src2 + atmp9 * src3);
202  final float dst6 = (atmp3 * src0 + atmp6 * src1 + atmp11 * src3)
203  - (atmp2 * src0 + atmp7 * src1 + atmp10 * src3);
204  final float dst7 = (atmp4 * src0 + atmp9 * src1 + atmp10 * src2)
205  - (atmp5 * src0 + atmp8 * src1 + atmp11 * src2);
206 
207  // calculate pairs for second 8 elements (cofactors)
208  final float btmp0 = src2 * src7;
209  final float btmp1 = src3 * src6;
210  final float btmp2 = src1 * src7;
211  final float btmp3 = src3 * src5;
212  final float btmp4 = src1 * src6;
213  final float btmp5 = src2 * src5;
214  final float btmp6 = src0 * src7;
215  final float btmp7 = src3 * src4;
216  final float btmp8 = src0 * src6;
217  final float btmp9 = src2 * src4;
218  final float btmp10 = src0 * src5;
219  final float btmp11 = src1 * src4;
220 
221  // calculate second 8 elements (cofactors)
222  final float dst8 = (btmp0 * src13 + btmp3 * src14 + btmp4 * src15)
223  - (btmp1 * src13 + btmp2 * src14 + btmp5 * src15);
224  final float dst9 = (btmp1 * src12 + btmp6 * src14 + btmp9 * src15)
225  - (btmp0 * src12 + btmp7 * src14 + btmp8 * src15);
226  final float dst10 = (btmp2 * src12 + btmp7 * src13 + btmp10 * src15)
227  - (btmp3 * src12 + btmp6 * src13 + btmp11 * src15);
228  final float dst11 = (btmp5 * src12 + btmp8 * src13 + btmp11 * src14)
229  - (btmp4 * src12 + btmp9 * src13 + btmp10 * src14);
230  final float dst12 = (btmp2 * src10 + btmp5 * src11 + btmp1 * src9)
231  - (btmp4 * src11 + btmp0 * src9 + btmp3 * src10);
232  final float dst13 = (btmp8 * src11 + btmp0 * src8 + btmp7 * src10)
233  - (btmp6 * src10 + btmp9 * src11 + btmp1 * src8);
234  final float dst14 = (btmp6 * src9 + btmp11 * src11 + btmp3 * src8)
235  - (btmp10 * src11 + btmp2 * src8 + btmp7 * src9);
236  final float dst15 = (btmp10 * src10 + btmp4 * src8 + btmp9 * src9)
237  - (btmp8 * src9 + btmp11 * src10 + btmp5 * src8);
238 
239  // calculate determinant
240  final float det = src0 * dst0 + src1 * dst1 + src2 * dst2 + src3 * dst3;
241 
242  if (det == 0.0f) {
243  return false;
244  }
245 
246  // calculate matrix inverse
247  final float invdet = 1.0f / det;
248  mInv[mInvOffset] = dst0 * invdet;
249  mInv[1 + mInvOffset] = dst1 * invdet;
250  mInv[2 + mInvOffset] = dst2 * invdet;
251  mInv[3 + mInvOffset] = dst3 * invdet;
252 
253  mInv[4 + mInvOffset] = dst4 * invdet;
254  mInv[5 + mInvOffset] = dst5 * invdet;
255  mInv[6 + mInvOffset] = dst6 * invdet;
256  mInv[7 + mInvOffset] = dst7 * invdet;
257 
258  mInv[8 + mInvOffset] = dst8 * invdet;
259  mInv[9 + mInvOffset] = dst9 * invdet;
260  mInv[10 + mInvOffset] = dst10 * invdet;
261  mInv[11 + mInvOffset] = dst11 * invdet;
262 
263  mInv[12 + mInvOffset] = dst12 * invdet;
264  mInv[13 + mInvOffset] = dst13 * invdet;
265  mInv[14 + mInvOffset] = dst14 * invdet;
266  mInv[15 + mInvOffset] = dst15 * invdet;
267 
268  return true;
269  }

334  public static void frustumM(float[] m, int offset, float left, float right, float bottom, float top, float near,
335  float far) {
336  if (left == right) {
337  throw new IllegalArgumentException("left == right");
338  }
339  if (top == bottom) {
340  throw new IllegalArgumentException("top == bottom");
341  }
342  if (near == far) {
343  throw new IllegalArgumentException("near == far");
344  }
345  if (near <= 0.0f) {
346  throw new IllegalArgumentException("near <= 0.0f");
347  }
348  if (far <= 0.0f) {
349  throw new IllegalArgumentException("far <= 0.0f");
350  }
351  final float r_width = 1.0f / (right - left);
352  final float r_height = 1.0f / (top - bottom);
353  final float r_depth = 1.0f / (near - far);
354  final float x = 2.0f * (near * r_width);
355  final float y = 2.0f * (near * r_height);
356  final float A = 2.0f * ((right + left) * r_width);
357  final float B = (top + bottom) * r_height;
358  final float C = (far + near) * r_depth;
359  final float D = 2.0f * (far * near * r_depth);
360  m[offset] = x;
361  m[offset + 5] = y;
362  m[offset + 8] = A;
363  m[offset + 9] = B;
364  m[offset + 10] = C;
365  m[offset + 14] = D;
366  m[offset + 11] = -1.0f;
367  m[offset + 1] = 0.0f;
368  m[offset + 2] = 0.0f;
369  m[offset + 3] = 0.0f;
370  m[offset + 4] = 0.0f;
371  m[offset + 6] = 0.0f;
372  m[offset + 7] = 0.0f;
373  m[offset + 12] = 0.0f;
374  m[offset + 13] = 0.0f;
375  m[offset + 15] = 0.0f;
